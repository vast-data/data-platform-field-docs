    connection_type,
    s3_bucket_name,
    s3_multipart_upload_id,
    s3_version_id,
    s3_version_phandle,
    s3_source_object,
    smb_ads_name,
    smb_create_action,
    smb_create_option,
    smb_create_disposition,
    smb_delete_on_close,
    smb_info_type,
    smb_info_class,
    smb_rename_struct,
    num_ops,
    num_bytes,
    time_str
FROM 
    "vast-audit-log-bucket|vast_audit_log_schema".vast_audit_log_table
WHERE
    view_path like '/csnow%' 
    and protocol <> 'NDB'
ORDER BY
    time DESC
LIMIT 10
